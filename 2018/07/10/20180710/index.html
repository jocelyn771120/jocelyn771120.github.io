0x2 02_angr_find_condition

explore()的参数相当于对到达路径起了约束作用。
find参数：程序会在找到对应find条件的states放在一个found list里面
avoid参数：与find类似，找到相应的states之后放在avoided lisit里面

import angr,sys

def main():

    pro=angr.Project("./02_angr_find_condition")

    entry_addr=pro.factory.entry_state()

    simulation=pro.factory.simgr(entry_addr)

    print "!"

    def success(state):
        stdout_res=state.posix.dumps(sys.stdout.fileno())
        return 'Good Job.' in stdout_res

    def fail(state):
        stdout_res=state.posix.dumps(sys.stdout.fileno())
        return 'Try again.' in stdout_res

    simulation.explore(find=success,avoid=fail)

    if simulation.found:
        solution=simulation.found[0]
        print "1"
        print solution.posix.dumps(sys.stdin.fileno())

    else:
        print "Error"

if __name__ == "__main__":

    main()
